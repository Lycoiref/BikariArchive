void MBEXRateCal::Roll()
{
    int Health = ui->spin_Health->value();
    int Times = ui->spin_Times->value();
    if (Times == 0) {
        ui->double_KillRate->setValue(0);
        return;
    }
    int KillTimes = 0;
    double AllDamage;

    srand((unsigned int)time(NULL));
    for (int t = 0; t < Times; t++) {
        AllDamage = 0;
        for (auto buki : ParamList) {
            int damage = buki->getDamage();
            int hit = buki->getHit();
            double critRate = buki->getCritRate();
            double critDamage = buki->getCritDamage();
            for (int i = 0; i < hit; i++) {
                AllDamage += damage * ((rand() % 10001 < critRate * 100) ? (critDamage / 100) : 1);
            }
        }
        if (AllDamage >= Health) KillTimes++;
    }
    ui->double_KillRate->setValue(KillTimes / (double)Times);

    int Damage_Zero = 0;
    int Damage_Exp = 0;
    int Damage_Mega = 0;
    for (auto buki : ParamList) {
        int Damage = buki->getDamage() * buki->getHit();
        double critRate = buki->getCritRate() / 100;
        double critDamage = buki->getCritDamage() / 100;
        Damage_Zero += Damage;
        Damage_Exp += Damage * (critRate * critDamage + (1 - critRate));
        Damage_Mega += Damage * critDamage;
    }
    ui->text_Output->append(QString("基础伤害：%1").arg(Damage_Zero));
    ui->text_Output->append(QString("期望伤害：%1").arg(Damage_Exp));
    ui->text_Output->append(QString("暴击伤害：%1").arg(Damage_Mega));
}
